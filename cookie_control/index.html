Cookie control

At work, I was tasked to have a modal pop up once per session, for each vistor.
I wound up doing some googling and found 'jquery-cookie.' I find it amazing that almost anything you can think of exists out there, and if it doesn't with some tinkering and some well placed questions on 'Stack Overflow' you can get it to work!! 

 1 <script type="text/javascript">
 2         var j = jQuery.noConflict();
 3 
 4         j(document).ready(function() {
 5             function openFancybox() {
 6                 setTimeout(function() {
 7                     j("#clickbanner").trigger('click');
 8                 }, 500);
 9 
10             };
11 
12             var visited = j.cookie('visited');
13             if (visited == 'yes') {
14                 return false;
15             } else {
16                 openFancybox();
17             };
18             j.cookie('visited', 'yes', {
19                 expires: 1
20             });
21 
22             j("#clickbanner").click(function() {
23                 j.fancybox({

28                 });
29                 return false;
30             });
31         });
